".*"			return TOKEN_STRING
;			return TOKEN_SEMICOLON
|			return TOKEN_PIPE
>			return TOKEN_RREDIR
>>			return TOKEN_DRREDIR
<			return TOKEN_LREDIR
<<			return TOKEN_DLREDIR
<<<			return TOKEN_TLREDIR
||			return TOKEN_ORIF
&&			return TOKEN_ANDIF

EXP_COMPLETE_L3		::=	
EXP_COMPLETE_L2		::=	EXP_COMPLETE_L1 (EXP_COMPOSE (EXP_COMPLETE_L2)?)?
EXP_COMPLETE_L1		::=	EXP_COMPLETE ((EXP_ANDIF | EXP_ORIF) EXP_COMPLETE_L1)?
EXP_COMPLETE		::=	EXP_SIMPLE (EXP_PIPE EXP_COMPLETE)?
EXP_SIMPLE		::=	(TOKEN_STRING | REDIR)*

EXP_COMPOSE		::=	TOKEN_SEMICOLON
EXP_PIPE		::=	TOKEN_PIPE
EXP_ANDIF		::=	TOKEN_ANDIF
EXP_ORIF		::=	TOKEN_ORIF

REDIR			::=	RREDIR | DRREDIR | LREDIR | DLREDIR | TLREDIR
RREDIR			::=	TOKEN_RREDIR TOKEN_STRING
DRREDIR			::=	TOKEN_DRREDIR TOKEN_STRING
LREDIR			::=	TOKEN_LREDIR TOKEN_STRING
DLREDIR			::=	TOKEN_DLREDIR TOKEN_STRING
TLREDIR			::=	TOKEN_TLREDIR TOKEN_STRING
